xh<-function(sigma_y,omega,sigma_n, c, a) {
  xi<-(sqrt(2)*c*omega^2*(sigma_y^2 + omega^2)) / ( abs(a)*sigma_n )
  y<-HypergU(-0.5, 0, xi^2)/xi
  g<-data.frame(xi,y)
  return(g)}

#note if we have too much error, we can get negative fitnesses, but optimum is always positive
#there is print function to indicate when get neg fitnesses. Best avoided with large omegasq

y_opt <- function(omegasq, af, n, c) (-af * n + sqrt((af * n)^2 + 4 * c^2 * omegasq)) / (2 * c)

fitness_for_any_y <- function(y, af, n, c, omegasq) {
  (1 / sqrt(2 * pi * omegasq)) * exp(-y^2 / (2 * omegasq)) * (af * n + c * y)}


#add error to an individual n and compute the fitness.summary is if we want mean stats of list of y and fitness
errorf<-function(n, omegasq, af,  c, summary) {
accumulator<-list()
for (j in 1:length(n)){
y<-rnorm(100, y_opt(n=n[j], omegasq=omegasq, af=af, c=c), sqrt(vge))
fitnesses<-fitness_for_any_y(y, af, n=n[j], c, omegasq)
if(sum(fitnesses<0)>0) print(sum(fitnesses<0)) #ensure in region with no negative fitnesses
accumulator[[j]]<-data.frame(y,fitnesses)}
res <- do.call(rbind, accumulator)
res <- as.data.frame(res)
seldiff<-cov(res$y,res$fitnesses)/mean(res$fitnesses)
stddseldiff<-seldiff/sqrt(var(res$y))
yvar<-var(res$y)
ybar<-mean(res$y) 
heritability<-vg/yvar
Ymaxfit<-res$y[which(res$fitnesses==max(res$fitnesses))][1]
Ymeanfit<-res$y[which(abs(res$fitnesses-mean(res$fitnesses))==min(abs(res$fitnesses-mean(res$fitnesses))))]

if(summary==1) return(data.frame(ybar, yvar, seldiff, stddseldiff, heritability, Ymaxfit,Ymeanfit)) 

if(summary==2) return(res)#this was to compute fitness plot and meanfitness
}

#function written by perplexity
# Given parameters:
# theta1, theta2 : means of G1 and G2
# omega1, omega2 : standard deviations (widths) of G1 and G2
# k              : additive constant

mix_mean_var <- function(theta1, theta2, omega1, omega2, k) {
  # Product Gaussian parameters
  omega12_sq <- (omega1^2 * omega2^2) / (omega1^2 + omega2^2)
  omega12    <- sqrt(omega12_sq)
  theta12    <- (theta1 * omega2^2 + theta2 * omega1^2) / (omega1^2 + omega2^2)
  
  # Amplitude factor from product G1 * G2
  C <- exp(-(theta1 - theta2)^2 / (2 * (omega1^2 + omega2^2))) *
       sqrt(1 / (omega1^2 + omega2^2))
  
  # Integrals (unnormalized weights)
  A1 <- k * sqrt(2 * pi) * omega1
  A2 <- C * sqrt(2 * pi) * omega12
  
  # Mixture weights
  w1 <- A1 / (A1 + A2)
  w2 <- A2 / (A1 + A2)
  
  # Mean of the normalized mixture
  mu <- w1 * theta1 + w2 * theta12
  
  # Variance of the normalized mixture
  var <- w1 * (omega1^2 + (theta1   - mu)^2) +
         w2 * (omega12^2 + (theta12 - mu)^2)
  
  list(mean = mu, var = var, sd = sqrt(var),
       w1 = w1, w2 = w2,
       theta12 = theta12, omega12 = omega12)
}
